L, C = map(int, input().split())
alpha = sorted(input().split())  #1

def is_password(arr):
  cnt_vow = 0
  for i in arr:
    if i == 'a' or i == 'e' or i == 'i' or i == 'o' or i == 'u':
      cnt_vow += 1
  if cnt_vow >= 1 and L - cnt_vow >= 2:
    print(''.join(arr))  #2

def dfs(v, arr):
  if len(arr) == L:
    is_password(arr)
    return
  if v == C:
    return
  arr.append(alpha[v])
  dfs(v+1, arr)
  arr.pop()
  dfs(v+1, arr)
